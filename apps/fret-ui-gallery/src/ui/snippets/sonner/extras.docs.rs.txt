// region: example
use fret::app::UiCxActionsExt as _;
use fret::{UiChild, AppComponentCx};
use fret_ui_shadcn::{facade as shadcn, prelude::*};

const CMD_TOAST_ACTION: &str = "examples.sonner.action";

pub fn render(cx: &mut AppComponentCx<'_>) -> impl UiChild + use<> {
    let sonner = shadcn::Sonner::global(&mut *cx.app);

    let action = {
        let sonner = sonner.clone();
        shadcn::Button::new("Action")
            .variant(shadcn::ButtonVariant::Outline)
            .on_activate(cx.actions().listen(move |host, action_cx| {
                sonner.toast_message(
                    host,
                    action_cx.window,
                    "Event has been created",
                    shadcn::ToastMessageOptions::new().action_id("Undo", CMD_TOAST_ACTION),
                );
                host.request_redraw(action_cx.window);
            }))
            .into_element(cx)
    };

    let cancel = {
        let sonner = sonner.clone();
        shadcn::Button::new("Cancel")
            .variant(shadcn::ButtonVariant::Outline)
            .on_activate(cx.actions().listen(move |host, action_cx| {
                sonner.toast_message(
                    host,
                    action_cx.window,
                    "Event has been created",
                    shadcn::ToastMessageOptions::new().cancel_id("Cancel", CMD_TOAST_ACTION),
                );
                host.request_redraw(action_cx.window);
            }))
            .into_element(cx)
    };

    let swipe = shadcn::Button::new("Swipe Dismiss Toast")
        .variant(shadcn::ButtonVariant::Outline)
        .on_activate(cx.actions().listen(move |host, action_cx| {
            sonner.toast(
                host,
                action_cx.window,
                shadcn::ToastRequest::new("Swipe to dismiss")
                    .description("Drag up to dismiss (pinned)")
                    .duration(None)
                    .dismissible(true),
            );
            host.request_redraw(action_cx.window);
        }))
        .into_element(cx);

    let controls = ui::h_flex(move |_cx| vec![action, cancel, swipe])
        .gap(Space::N2)
        .items_center()
        .wrap()
        .layout(LayoutRefinement::default().w_full().min_w_0())
        .into_element(cx);

    let toaster = shadcn::Toaster::new().into_element(cx);

    ui::v_flex(move |_cx| vec![controls, toaster])
        .gap(Space::N3)
        .items_start()
        .layout(LayoutRefinement::default().w_full().min_w_0())
        .into_element(cx)
}
// endregion: example
